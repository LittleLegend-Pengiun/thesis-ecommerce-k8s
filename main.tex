\expandafter{\language@yaml}
{
  keywords={true,false,null,y,n},

  sensitive=false,

  morestring=[b]",

                {>}{{\textcolor{red}\textgreater}}1     
                {|}{{\textcolor{red}\textbar}}1 
                {\ -\ }{{\mdseries\ -\ }}3,
}
